མ་མཆིས་པ།་།དེ་ལྟར་ཉིད་ཀྱི་རྩལ་གྱིས་འབད་པར་མཛོད།་།ཅེས་སོ།་དུད་འགྲོའི་སྡུག་བསྔལ་བསམ་པ་ནི།་དུད་འགྲོ་རྣམས་མཐུ་ཆེ་བས་ཆུང་བ་རྣམས་གསོད་པ་དང་།་ལྷ་དང་མིའི་ཡོ་བྱད་དུ་གྱུར་ཏེ་རང་ལ་རང་དབང་མེད་པར་གཞན་
གྱི་དབང་དུ་གྱུར་པས་གསོད་པ་དང་བརྡེག་པ་དང་གནོད་པ་བྱེད་ལ།་ལྷ་དང་མི་དང་ལྷན་ཅིག་རྒྱུ་བས་གནས་གཞན་མེད་པར་སའི་དངོས་གཞི་ལས་བཤད་དོ།་།མཛོད་འགྲེལ་ལས་ནི།་དུད་འགྲོ་ནི་ཐང་དང་ཆུ་དང་ནམ་མཁའི་སྤྱོད་ཡུལ་བ་
དག་ཡིན་ནོ།་།དེ་དག་གི་གནས་ཀྱི་རྩ་བ་ནི་རྒྱ་མཚོ་ཆེན་པོ་ཡིན་ཏེ་གཞན་ནི་དེ་ལས་འཕྲོས་པ་དག་ཡིན་ནོ།་།ཞེས་གསུངས་སོ།་།བཤེས་སྤྲིངས་ལས་ཀྱང་།་དུད་འགྲོའི་སྐྱེ་གནས་ན་ཡང་གསོད་པ་དང་།་།བཅིང་དང་བརྡེག་
སོགས་སྡུག་བསྔལ་སྣ་ཚོགས་པ།་།ཞི་འགྱུར་དགེ་བ་སྤངས་པ་རྣམས་ལ་ནི།་།གཅིག་ལ་གཅིག་ཟ་ཤིན་ཏུ་མི་བཟད་པ།་།ཁ་ཅིག་མུ་ཏིག་བལ་དང་རུས་པ་དང་།་།ཤ་དང་པགས་པའི་ཆེད་དུ་འཆི་བར་འགྱུར།་།དབང་མེད་
གཞན་དག་རྡོག་པ་ལག་པ་དང་།་།ལྕག་དང་ལྕགས་ཀྱུ་གདབ་པས་བཏབ་སྟེ་བཀོལ།་།ཞེས་སོ།་།དེ་ལ་ཚིགས་བཅད་དང་པོས་སྤྱི་དང་།་གཉིས་པས་ཁྱད་པར་གྱི་སྡུག་བསྔལ་བསྟན་ཏོ།་།བརྡེག་སོགས་ཞེས་པའི་སོགས་ཀྱི་
སྒྲས་ནི་བཀོལ་བ་དང་སྣ་དབུག་པ་སོགས་བསྡུས་ཏེ།་དེ་ནི་མི་དང་མི་མ་ཡིན་པ་རྣམས་ཀྱིས་གསོད་པ་སོགས་བྱེད་པའི་དབང་དུ་མཛད་དོ།་།གཅིག་ལ་གཅིག་ཟ་ཞེས་པ་ནི་དུད་འགྲོ་རིགས་མཐུན་གྱི་གནོད་པའི་དབང་དུ་མཛད་པ་སྟེ།་
